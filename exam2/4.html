function formatDateTime(date) { if (typeof date === 'undefined') { return '请输入日期'; } else if (!(date instanceof Date)) { return '输入的不是日期类型'; } year = date.getFullYear(); month = date.getMonth(); day = date.getDate(); hour = date.getHours(); minute = date.getMinutes(); ret = year + '-' + month + '-' + day + ' ' + hour + ':' + minute; return ret; }
